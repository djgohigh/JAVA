	public static void searchBookPublishingYear(String Word1, String Word2, Book BookList[]) {
		StringTokenizer searchDate = null;
		int StartDate[]=new int[3];
		int EndDate[]=new int[3];
		int DB_Date[] = new int [3];
		int i = 0;
		
		searchDate = new StringTokenizer(Word1, "-");
		while(searchDate.hasMoreTokens()) {
			StartDate[i]=Integer.parseInt(searchDate.nextToken());
			i++;
		}
		
		i = 0;
		searchDate = new StringTokenizer(Word2, "-");
		while(searchDate.hasMoreTokens()) {
			EndDate[i]=Integer.parseInt(searchDate.nextToken());
			i++;
		}
		
		System.out.print("시작 년월일: ");
		for(i = 0; i<3; i++) {
			System.out.printf("%d", StartDate[i]);
		}
		System.out.printf("\n");
		
		System.out.print("끝 년월일: ");
		for(i = 0; i<3; i++) {
			System.out.printf("%d", EndDate[i]);
		}
		System.out.printf("\n");
		
		for(i = 0; i<100; i++) {
			if(BookList[i]==null) {
				break;
			} else {
				String DB_BookPublishingYear=BookList[i].getBookPublishingYear();
				searchDate = new StringTokenizer(DB_BookPublishingYear, "-");
				
				int j=0;
				while(searchDate.hasMoreTokens()) {
					DB_Date[j]=Integer.parseInt(searchDate.nextToken());
					j++;
				}
				if(StartDate[0] < DB_Date[0] && EndDate[0] > DB_Date[0]) {
					BookList[i].infoBook();
					System.out.printf("\n");
				}
				if(StartDate[0] == DB_Date[0] && EndDate[0] > DB_Date[0]) {
					if(StartDate[1] < DB_Date[1]) {
						BookList[i].infoBook();
						System.out.printf("\n");
						}
					if(StartDate[1] == DB_Date[1]) {
						if(StartDate[2] <= DB_Date[1]) {
							BookList[i].infoBook();
							System.out.printf("\n");
						}
					}
				}
				if(StartDate[0] < DB_Date[0] && EndDate[0] == DB_Date[0] ) {
					if(EndDate[1] > DB_Date[1]) {
						BookList[i].infoBook();
						System.out.printf("\n");
						}
					if(EndDate[1] == DB_Date[1]) {
						if(EndDate[2] >= DB_Date[1]) {
							BookList[i].infoBook();
							System.out.printf("\n");
						}
					}
				}
				if(StartDate[0] == DB_Date[0] && EndDate[1] == DB_Date[1] ) {
					if(StartDate[2] <= DB_Date[2] && EndDate[2] >= DB_Date[2]) {
						BookList[i].infoBook();
						System.out.printf("\n");
					}
				}
			}
		}
	}

// Book

package SearchLibrary;

public class Book {
	private String BookName;
	private String Author;
	private String Publisher;
	private String PublishingYear;
	private int Page;
	private String LibraryName;
	private int ID;
	private String StorePlace;
	private int CheckOutNum;
	private boolean isCheckOut;
	
	public Book() {
	}
	
	public Book(String BookName, String Author, String Publisher, String PublishingYear, 
			    int Page, String LibraryName, int ID, String StorePlace, 
			    int CheckOutNum, boolean isCheckOut) {
		this.BookName = BookName;
		this.Author = Author;
		this.Publisher = Publisher;
		this.PublishingYear = PublishingYear;
		this.Page = Page;
		this.LibraryName = LibraryName;
		this.ID=ID;
		this.StorePlace=StorePlace;
		this.CheckOutNum=CheckOutNum;
		this.isCheckOut=isCheckOut;
	}
	
	public String getBookName() {
		return BookName;
	}
	
	public String getBookAuthor() {
		return Author;
	}
	
	public String getBookPublisher() {
		return Publisher;
	}
	
	public String getBookStorePlace() {
		return StorePlace;
	}
	
	public String getBookPublishingYear() {
		return PublishingYear;
	}
	
	public void infoBook() {
		System.out.printf("제목: %s \n", BookName);
		System.out.printf("작가명: %s \n", Author);
		System.out.printf("출판사: %s \n", Publisher);
		System.out.printf("출판일: %s \n", PublishingYear);
		System.out.printf("페이지 수 %d \n", Page);
		System.out.printf("소장도서관: %s \n", LibraryName);
		System.out.printf("아이디: %d \n", ID);
		System.out.printf("분류: %s \n", StorePlace);
		System.out.printf("대출횟수: %d \n", CheckOutNum);
		System.out.printf("대충가능여부: %b \n", isCheckOut);
		System.out.printf("\n");
	}
}
